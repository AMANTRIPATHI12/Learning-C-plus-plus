class BankAccount{
    private:
        string accountNumber;
        double balance;

    public:
        BankAccount (string accNum, double bal){
            accountNumber = accNum;
            balance = bal;
        }
        // Getters
        string getName()const{
            return accountNumber;
        }

        double getMarks()const{
            return balance;
        }

        void deposit(double amount){
            if(amount > 0){
                balance += amount;
            }
        }

        void withdraw(double amount){
            if(amount > 0 && amount <= balance){
                balance -= amount;
            }
        }

        // display
        void display()const{
            cout << "sufficient balance : " << balance << endl;
        }
};

class Student{
    private:
        string name;
        int roll_number;
        double marks;

    public:
        Student(string n, int r, double m){
            name = n;
            roll_number = r;
            marks = m;
        }
        // Getters
        string getName()const{
            return name;
        }

        int getRollNumber()const{
            return roll_number;
        }

        double getMarks()const{
            return marks;
        }
        // display
        void display()const{
            cout << "Roll no : " << roll_number << ", Name : " << name << ", Marks : " << marks << endl;
        }
};

// function to find topper (using array of objects)
int findTopper(Student students[], int size){
    int topperIndex = 0;
    for(int i = 0; i < size; i++){
        if(students[i].getMarks() > students[topperIndex].getMarks()){
            topperIndex = i;
        }
    }
    return topperIndex;
}

int main(){
    cout << "Hello multiverse" << endl;
    Student s1("Amar",101,95.2);
    Student s2("Akhbar",102,96.2);
    Student s3("Anthany",103,97.2);

    Student batch[] = {s1,s2,s3};
    int size = sizeof(batch) / sizeof(batch[0]);
    int top = findTopper(batch , size);
    cout << "Topper : ";
    batch[top].display();
    for(int i = 0; i < size - 1; i++){
        for(int j = i + 1; j < size; j++){
            if(batch[i].getMarks() < batch[j].getMarks()){
                swap(batch[i],batch[j]);
            }
        }
        
    }

    cout << "New sorted list" << endl;

    for(int i = 0; i < size; i++){
        batch[i].display();
    }

class Rectangle{
    private:
        double length;
        double width;
    public:
        // This is a constructor which gets called when a object gets created.
        Rectangle(double l,double w){
            length = l;
            width = w;
        }

        // Member fuctions
        double area(){
            return length * width;
        }

        double perimeter() {
            return 2 * (length + width);
        }

        // Setter (mutator)
        void setLength(double l){
            if(l <= 0){
                cout << "Enter value greater than 0" << endl;
            }
            else{
                length = l;
            }
        }
        
        // getter (accesser)
        double getLength(){
            return length;
        }
};
